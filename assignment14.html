df <- read_csv('twitterdata.csv')

ts_plot(df, "hours") +
  labs(x = NULL, y = NULL,
       title = "Frequency of tweets by time",
       subtitle = paste0(format(min(df$created_at), "%d %B %Y"), " to ", format(max(df$created_at),"%d %B %Y")),
       caption = "Data collected from Twitter's REST API via rtweet") +
  theme_minimal()

df %>% 
  unnest_tokens(mentions, text, "tweets", to_lower = FALSE) %>%
  filter(str_detect(mentions, "^@")) %>%  
  count(mentions, sort = TRUE) %>%
  head(10) %>% 
  ggplot(aes(x = n, y = reorder(mentions, n)))+
  geom_col()+
  labs(title = 'Most @ mentions',x = '# of mentions', y = '')

df %>% 
  filter(created_at>='2022-11-14') %>% 
  unnest_tokens(output = hashtag, input = text, token = "tweets") %>%
  filter(!hashtag %in% c("#dragonflight"), str_detect(hashtag, "^#")) %>% 
  count(hashtag, sort = TRUE) %>%
  head(5) %>% 
  ggplot(aes(x = n, y = reorder(hashtag, n)))+
  geom_col()+
  labs(x = 'Frequency', y = '', title = 'other # mentions')

df %>%
  mutate(emoji = ji_extract_all(text)) %>%
  unnest(cols = c(emoji)) %>%
  count(emoji, sort = TRUE) %>%
  head(5) %>% 
  ggplot(aes(y=reorder(emoji,n), x=n)) +
  geom_col()+
  theme(axis.text.y = element_text(size = 40))+
  labs(title = 'Top Emoji in the tweets', y = '')

df %>%
    unnest_tokens(input = text, output = word) %>% 
    anti_join(get_stopwords()) %>% 
    count(word, sort = TRUE) %>%
    inner_join(get_sentiments("nrc")) %>%
    filter(!is.na(sentiment)) %>%
    count(sentiment, sort = TRUE) %>% 
    mutate(n = n/sum(n)) %>% 
    ggplot(aes(y= reorder(sentiment, n), x = n))+geom_col()+
    labs(y='Relative Frequency', x ='', title = "sentiment frequencies")
